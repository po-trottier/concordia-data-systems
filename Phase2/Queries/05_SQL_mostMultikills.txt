CREATE OR REPLACE VIEW multikill_champion ASSELECT championId, doubleKills, tripleKills, quadraKills, pentaKills, doubleKills+tripleKills+quadraKills+pentaKills as totalMultiKillFROM participants_filtered JOIN stats_filtered ON participants_filtered.gameId = stats_filtered.gameId;SELECT championId, MAX(totalMultiKill) AS multi_killzFROM multikill_championGROUP BY championId ORDER BY multi_killz DESCLIMIT 10;
