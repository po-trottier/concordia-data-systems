CREATE OR REPLACE VIEW winning_team ASSELECT DISTINCT *FROM stats_filterdWHERE win = 'true';CREATE OR REPLACE VIEW losing_team ASSELECT DISTINCT *FROM stats_filterdWHERE win = 'false';CREATE OR REPLACE VIEW something_about_CC ASSELECT winning_team.gameId, SUM(winning_team.timeCCingOthers) AS "CC_Winning_Team", SUM(losing_team.timeCCingOthers) AS "CC_Losing_Team"FROM winning_team JOIN losing_team ON winning_team.gameId = losing_team.gameIdGROUP BY winning_team.gameId;SELECT AVG("CC_Winning_Team") AS Winning_Team_Average, AVG("CC_Losing_Team") AS Losing_Team_AverageFROM something_about_CC
