[{$unwind: "$participants"}, {$group: {
  _id: {
    "gameId": "$gameId",
    "teamId": "$participants.teamId"
  },
  timeCCing: {
    $sum: "$participants.stats.totalTimeCrowdControlDealt"
  },
  win: {
    $first: "$participants.stats.win"
  }
}}, {$sort: {
  "_id.gameId": 1,
  "timeCCing": -1
}}, {$group: {
  _id: "$_id.gameId",
  timeCCingWin: {
    $first: "$win"
  }
}}, {$group: {
  "_id": {
    "timeCCingWin": "$timeCCingWin"
  },
  "count": {
    $sum: 1
  }
}}, {$sort: {
  "_id.timeCCingWin": -1
}}, {$group: {
  _id: null,
  moreTimeCCingWin: {
    $first: "$count"
  },
  lessTimeCCingWin: {
    $last: "$count"
  },
  total: {
    $sum: "$count"
  }
}}, {$project: {
  moreTimeCCingWinPercentage: {
    $divide: ["$moreTimeCCingWin", "$total"]
  }
}}, {$project: {
  betterVisionWinPercentage: {
    $multiply: ["$moreTimeCCingWinPercentage", 100]
  }
}}]
