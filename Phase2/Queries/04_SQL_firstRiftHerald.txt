CREATE OR REPLACE VIEW rift_herald_win ASSELECT COUNT(gameId) as rift_winning_team FROM teams_filteredWHERE win = 'Win' AND firstRiftHerald = 'true';CREATE OR REPLACE VIEW rift_herald_loss AS SELECT COUNT(gameId) as rift_losing_teamFROM teams_filteredWHERE win = 'Fail' AND firstRiftHerald = 'true';SELECT rift_winning_team, rift_losing_teamFROM rift_herald_win, rift_herald_loss;
