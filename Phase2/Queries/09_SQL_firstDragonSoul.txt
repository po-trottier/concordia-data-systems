CREATE OR REPLACE VIEW dragon_soul_win ASSELECT COUNT(gameId) as dragon_winning_team FROM teams_filteredWHERE win = 'Win'  AND dragonKills >= 4;CREATE OR REPLACE VIEW dragon_soul_lose ASSELECT COUNT(gameId) as dragon_losing_team FROM teams_filteredWHERE win = 'Fail' AND dragonKills >= 4;   SELECT dragon_winning_team AS dragon_soul_winning_teams, dragon_losing_team AS dragon_soul_losing_teamsFROM dragon_soul_lose, dragon_soul_win;
