CREATE OR REPLACE VIEW teams_100 ASSELECT teamId, gameId, AVG(towerKills) AS avg_100FROM teams_filteredWHERE teamId = 100GROUP BY teamId, gameId;CREATE OR REPLACE VIEW teams_200 ASSELECT teamId, gameId, AVG(towerKills) AS avg_200FROM teams_filteredWHERE teamId = 200GROUP BY teamId, gameId;SELECT (AVG(avg_100) + AVG(avg_200)) / 2 AS average_tower_kills_per_teamFROM teams_100, teams_200;
